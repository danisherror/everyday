#include<stdio.h>
void main ()
{
        int i,j,n;
        printf("Enter the value of n for pattern\n");
        scanf("%d",&n);
        printf("Pattern is\n");
        for(i=1;i<=n;i++)
        {
                for(j=1;j<i;j++)
                {
                        printf(" ");
                }
                for(j=1;j<=n;j++)
                {
                        if(i==1|| j==1|| j==n||i==n)
                        {
                                printf("*");
                        }
                        else
                        {
                                printf(" ");
                        }
                }
                printf("\n");
        }
}
