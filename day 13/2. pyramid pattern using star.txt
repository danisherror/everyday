#include<iostream>
using namespace std;
int main()
{
        int n,j,i;
        cout<<"Enter the number of lines : ";
        cin>>n;
        for(i=0;i<=n;i++)
        {
                for(j=0;j<i;j++)
                {
                        cout<<"*";
                }
                cout<<endl;
        }
        return 0;
}
