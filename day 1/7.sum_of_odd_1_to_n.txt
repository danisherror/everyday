#include<stdio.h>
int main ()
{
        int i,n,sum=0;
        printf("enter the value of n \n ");
        scanf("%d",&n);
        for(i=1;i<=n;i++)
        {
                if(i%2==1)
                {
                        sum=sum +i;
                }
        }
        printf("sum of first odd numbers till %d is %d \n",n,sum);
        return 0;
}
